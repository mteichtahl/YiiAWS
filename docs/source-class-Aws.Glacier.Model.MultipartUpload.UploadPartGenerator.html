 70:      *
 71:      * @param string|resource|EntityBodyInterface $body The upload body

 73:      * @return UploadPart
 74:      * @throws RuntimeException if the body ends up being larger than 4GB
 75:      */
 76:     public static function createSingleUploadPart($body)
 77:     {
 78:         $generator = new static(EntityBody::factory($body), 4 * Size::GB);
 79:         if (count($generator) > 1) {
 80:             // @codeCoverageIgnoreStart
 81:             throw new RuntimeException('You cannot create a single upload that is larger than 4 GB.');
 82:             // @codeCoverageIgnoreEnd
 83:         }
 84:

 86:     }

 89:      * @param EntityBodyInterface $body     The upload body
 90:      * @param int                 $partSize The size of parts to split the upload into. Default is the 4GB max
 91:      *
 92:      * @throws InvalidArgumentException when the part size is invalid (i.e. not a power of 2 of 1MB)
 93:      * @throws InvalidArgumentException when the body is not seekable (must be able to rewind after calculating hashes)
 94:      */

 96:     {
 97:         $this->partSize = $partSize;
 98: 
 99:        // Make sure the part size is valid
100:         $validPartSizes = array_map(function ($value) {return pow(2, $value) * Size::MB;}, range(0, 12));
101:         if (!in_array($this->partSize, $validPartSizes)) {
102:             throw new InvalidArgumentException('The part size must be a megabyte multiplied by a power of 2 and no '
103:                 . 'greater than 4 gigabytes.');
104:         }
105: 
106:         // Validate body
107:         if (!$body->isSeekable()) {
108:             throw new InvalidArgumentException('The upload body must be seekable.');
109:         }
110: 
111:         $this->generateUploadParts($body);
112:     }

220:     /**
221:      * Performs the work of reading the body stream, creating tree hashes, and creating UploadPartContext objects
222:      *
223:      * @param EntityBodyInterface $body The body to create parts from
224:      */
225:     protected function generateUploadParts(EntityBodyInterface $body)
226:     {
227:         // Rewind the body stream
228:         $body->seek(0);
229: 
230:         // Initialize variables for tracking data for upload
231:         $uploadContext = new UploadPartContext($this->partSize, $body->ftell());
232: 
233:         // Read the data from the streamed body in 1MB chunks
234:         while ($data = $body->read(min($this->partSize, Size::MB))) {
235:             // Add data to the hashes and size calculations
236:             $uploadContext->addData($data);
237: 
238:             // If the upload part is complete, generate an upload object and reset the currently tracked upload data
239:             if ($uploadContext->isFull()) {
240:                 $this->updateTotals($uploadContext->generatePart());
241:                 $uploadContext = new UploadPartContext($this->partSize, $body->ftell());
242:             }
243:         }
244: 
245:         // Handle any leftover data
246:         if (!$uploadContext->isEmpty()) {
247:             $this->updateTotals($uploadContext->generatePart());
248:         }
249: 
250:         // Rewind the body stream
251:         $body->seek(0);
252:     }
253: 
254:     /**
255:      * Updated the upload helper running totals and tree hash with the data from a complete upload part
256:      *
257:      * @param UploadPart $part The newly completed upload part
258:      *
259:      * @throws OverflowException if the maximum number of allowed upload parts is exceeded
260:      */
261:     protected function updateTotals(UploadPart $part)
262:     {
263:         // Throw an exception if there are more parts than total allowed
264:         if ($part->getPartNumber() > self::MAX_NUM_PARTS) {
265:             // @codeCoverageIgnoreStart
266:             throw new OverflowException('An archive must be uploaded in ' . self::MAX_NUM_PARTS . ' parts or less.');
267:             // @codeCoverageIgnoreEnd
268:         }
269: 
270:         $this->uploadParts[] = $part;
271:         $this->archiveSize += $part->getSize();
272:     }
