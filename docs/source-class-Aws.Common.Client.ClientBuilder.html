229:      *

231:      * @throws InvalidArgumentException
232:      */
233:     public function build()
234:     {
235:         // Resolve config
236:         $config = Collection::fromConfig(
237:             $this->config,
238:             array_merge(self::$commonConfigDefaults, $this->configDefaults),
239:             (self::$commonConfigRequirements + $this->configRequirements)
240:         );
241: 
242:         // Set values from the service description
243:         $this->updateConfigFromDescription($config);
244: 
245:         // If no endpoint provider was explicitly set, the instantiate a default endpoint provider
246:         if (!$config->get(Options::ENDPOINT_PROVIDER)) {
247:             $config->set(Options::ENDPOINT_PROVIDER, $this->getDefaultEndpointProvider());
248:         }
249: 
250:         // If no base_url was explicitly set, then grab one using the default endpoint provider
251:         if (!$config->get(Options::BASE_URL)) {
252:             $this->addBaseUrlToConfig($config);
253:         }
254: 
255:         // Resolve credentials
256:         if (!$this->credentialsResolver) {
257:             $this->credentialsResolver = $this->getDefaultCredentialsResolver();
258:         }

260: 
261:         // Add other client resolvers, like exponential backoff
262:         if (!$this->hasBackoffOptionResolver()) {

264:         }
265:         $config->set(Options::RESOLVERS, $this->clientResolvers);
266: 
267:         // Determine service and class name
268:         $clientClass = 'Aws\Common\Client\DefaultClient';

270:             $serviceName = substr($this->clientNamespace, strrpos($this->clientNamespace, '\\') + 1);

451:     /**
452:      * Return an appropriate signature object for a a client based on a description

455:      * @param Collection         $config      Configuration options
456:      *
457:      * @throws InvalidArgumentException
458:      */
459:     protected function addSignature(ServiceDescription $description, Collection $config)
460:     {
461:         if (!($signature = $config->get(Options::SIGNATURE))) {
462:             if (!$description->getData('signatureVersion')) {
463:                 throw new InvalidArgumentException('The service description does not specify a signatureVersion');
464:             }
465:             switch ($description->getData('signatureVersion')) {
466:                 case 'v2':
467:                     $signature = new SignatureV2();
468:                     break;
469:                 case 'v3':
470:                     $signature = new SignatureV3();
471:                     break;
472:                 case 'v3https':
473:                     $signature = new SignatureV3Https();
474:                     break;
475:                 case 'v4':
476:                     $signature = new SignatureV4();
477:                     break;
478:             }
479:         }

481:         // Allow a custom service name or region value to be provided
482:         if ($signature instanceof EndpointSignatureInterface) {

484:                 $config->get(Options::SIGNATURE_SERVICE) ?: $description->getData('signingName')
485:             );
486:             $signature->setRegionName($config->get(Options::SIGNATURE_REGION));
487:         }
