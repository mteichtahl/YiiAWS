 26: /**
 27:  * Provides an interface for using Amazon DynamoDB as a session store by hooking
 28:  * into PHP's session handler hooks. Once registered, You may use the native
 29:  * `$_SESSION` superglobal and session functions, and the sessions will be
 30:  * stored automatically in DynamoDB. DynamoDB is a great session storage
 31:  * solution due to its speed, scalability, and fault tolerance.
 32:  *
 33:  * For maximum performance, we recommend that you keep the size of your sessions
 34:  * small. Items greater than 1KB require more throughput in DynamoDB. Also, in
 35:  * this version of the session handler, session locking is turned off by default
 36:  * since it can drive up latencies and costs under high traffic and poor session
 37:  * management, especially when using ajax. Only turn it on if you need it.
 38:  *
 39:  * By far, the most expensive operation is garbage collection. Therefore, we
 40:  * encourage you to carefully consider your session garbage collection strategy.
 41:  * We recommend that you change the `session.gc_probability` ini setting to 0 so
 42:  * that garbage collection is not triggered randomly. You should consider using
 43:  * a cron job or similar scheduling technique for triggering garbage collection
 44:  * at appropriate times.
 45:  */

 88:     /**
 89:      * Factory method to create a new DynamoDB Session Handler
 90:      *
 91:      * The configuration array accepts the following array keys and values:
 92:      * - locking_strategy:         Locking strategy for session locking logic
 93:      * - dynamodb_client:          Client for doing DynamoDB operations
 94:      * - table_name:               Name of the table in which to store sessions
 95:      * - hash_key:                 Name of the hash key in the sessions table
 96:      * - session_lifetime:         Lifetime of inactive sessions
 97:      * - consistent_read:          Use DynamoDB consistent reads for `GetItem`
 98:      * - automatic_gc:             Use PHP's auto garbage collection
 99:      * - gc_batch_size:            Batch size for garbage collection deletes
100:      * - max_lock_wait_time:       Max time to wait for lock acquisition
101:      * - min_lock_retry_microtime: Min time to wait between lock attempts
102:      * - max_lock_retry_microtime: Max time to wait between lock attempts
103:      *
104:      * @param array $config Configuration options
105:      *
106:      * @return SessionHandler
107:      */
108:     public static function factory(array $config = array())
109:     {
110:         // Setup session handler configuration and get the client
111:         $config = new SessionHandlerConfig($config);
112:         $client = $config->get('dynamodb_client');
113: 
114:         // Make sure locking strategy has been provided or provide a default
115:         $strategy = $config->get('locking_strategy');
116:         if (!($strategy instanceof LockingStrategyInterface)) {
117:             $factory  = new LockingStrategyFactory();
118:             $strategy = $factory->factory($strategy, $config);

121:         // Return an instance of the session handler
122:         return new static($client, $strategy, $config);

277:     /**
278:      * Read a session stored in DynamoDB
279:      *
280:      * Part of the standard PHP session handler interface
281:      *
282:      * @param string $id The session ID
283:      *
284:      * @return string The session data
285:      */
286:     public function read($id)
287:     {
288:         // PHP expects an empty string to be returned from this method if no
289:         // data is retrieved
290:         $this->dataRead = '';
291: 
292:         // Get session data using the selected locking strategy
293:         $item = $this->lockingStrategy->doRead($this->formatId($id));
294: 
295:         // Return the data if it is not expired. If it is expired, remove it
296:         if (isset($item['expires']) && isset($item['data'])) {
297:             $this->dataRead = $item['data'];
298:             if ($item['expires'] <= time()) {
299:                 $this->dataRead = '';
300:                 $this->destroy($id);
301:             }
302:         }
303: 
304:         return $this->dataRead;
305:     }

366:     /**
367:      * Performs garbage collection on the sessions stored in the DynamoDB table
368:      *
369:      * If triggering garbage collection manually, use this method. If your
370:      * garbage collection is triggered automatically by php (not recommended),
371:      * then use the `gc` method.
372:      */
373:     public function garbageCollect()
374:     {
375:         $deleteBatch = WriteRequestBatch::factory(
376:             $this->client,
377:             $this->config->get('gc_batch_size')
378:         );
379:         $tableName = $this->config->get('table_name');
380:         $expires   = (string) time();
381: 
382:         // Setup a scan table command for finding expired session items
383:         $tableScan = $this->client->getCommand('Scan', array(
384:             'TableName' => $tableName,
385:             'AttributesToGet' => array(
386:                 $this->config->get('hash_key')
387:             ),
388:             'ScanFilter' => array(
389:                 'expires' => array(
390:                     'ComparisonOperator' => 'LT',
391:                     'AttributeValueList' => array(
392:                         array(
393:                             'N' => $expires
394:                         )
395:                     ),
396:                 ),
397:                 'lock' => array(
398:                     'ComparisonOperator' => 'NULL',
399:                 )
400:             ),
401:             Ua::OPTION => Ua::SESSION
402:         ));
403: 
404:         // Perform scan and batch delete operations as needed
405:         foreach ($this->client->getIterator($tableScan) as $item) {
406:             // @codeCoverageIgnoreStart
407:             $key = array('HashKeyElement' => $item[$this->config->get('hash_key')]);
408:             $deleteBatch->add(new DeleteRequest($key, $tableName));
409:             // @codeCoverageIgnoreEnd
410:         }
411: 
412:         // Delete any remaining items
413:         $deleteBatch->flush();
414:     }
