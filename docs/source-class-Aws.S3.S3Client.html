251:     public static function isValidBucketName($bucket)
252:     {
253:         $bucketLen = strlen($bucket);
254:         if (!$bucket || $bucketLen < 3 || $bucketLen > 63
255:             // Cannot start or end with a '.'

257:             || $bucket[$bucketLen - 1] == '.'
