101:      */
102:     public function prepareData()
103:     {
104:         // Validate required options
105:         $options = Collection::fromConfig($this->data, array(
106:             'ttd' => '+1 hour',
107:             'key' => '^${filename}',
108:         ));
109: 
110:         // Format ttd option
111:         $ttd = $options['ttd'];
112:         $ttd = is_numeric($ttd) ? (int) $ttd : strtotime($ttd);
113:         $options->remove('ttd');
114: 
115:         // Save policy if passed in
116:         $rawPolicy = $options->get('policy');
117:         $options->remove('policy');
118: 
119:         // Setup policy document
120:         $policy = array(
121:             'expiration' => gmdate(DateFormat::ISO8601_S3, $ttd),
122:             'conditions' => array(
123:                 array('bucket' => $this->bucket),
124:             )
125:         );
126: 
127:         // Configure the endpoint/action
128:         $url = Url::factory($this->client->getBaseUrl());
129:         $url->setHost($this->bucket . '.' . $url->getHost());
130: 
131:         // Setup basic form
132:         $this->formAttributes = array(
133:             'action' => (string) $url,
134:             'method' => 'POST',
135:             'enctype' => 'multipart/form-data'
136:         );
137:         $this->formInputs = array(
138:             'AWSAccessKeyId' => $this->client->getCredentials()->getAccessKeyId()
139:         );
140: 
141:         // Add success action status
142:         $status = (int) $options->get('success_action_status');
143:         if ($status && in_array($status, array(200, 201, 204))) {
144:             $this->formInputs['success_action_status'] = (string) $status;
145:             $policy['conditions'][] = array(
146:                 'success_action_status' => (string) $status
147:             );
148:             $options->remove('success_action_status');
149:         }
150: 
151:         // Add other options
152:         foreach ($options as $key => $value) {
153:             $value = (string) $value;
154:             if ($value[0] === '^') {
155:                 $value = substr($value, 1);
156:                 $this->formInputs[$key] = $value;
157:                 $value = preg_replace('/\$\{(\w*)\}/', '', $value);
158:                 $policy['conditions'][] = array('starts-with', '$' . $key, $value);
159:             } else {
160:                 $this->formInputs[$key] = $value;
161:                 $policy['conditions'][] = array($key => $value);
162:             }
163:         }
164: 
165:         // Add policy
166:         $this->jsonPolicy = $rawPolicy ?: json_encode($policy);
167:         $jsonPolicy64 = base64_encode($this->jsonPolicy);
168:         $this->formInputs['policy'] = $jsonPolicy64;
169: 
170:         // Add signature

175:             true
176:         ));

178:         return $this;
