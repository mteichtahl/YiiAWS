382:     /**
383:      * Log a message at a priority
384:      *
385:      * @param  string   $message   Message to log
386:      * @param  integer  $priority  Priority of message
387:      * @param  mixed    $extras    Extra information to log in event
388:      * @return void
389:      * @throws Zend_Log_Exception
390:      */
391:     public function log($message, $priority, $extras = null)
392:     {
393:         // sanity checks
394:         if (empty($this->_writers)) {
395:             /** @see Zend_Log_Exception */
396:             require_once 'Zend/Log/Exception.php';
397:             throw new Zend_Log_Exception('No writers were added');
398:         }
399: 
400:         if (! isset($this->_priorities[$priority])) {
401:             /** @see Zend_Log_Exception */
402:             require_once 'Zend/Log/Exception.php';
403:             throw new Zend_Log_Exception('Bad log priority');
404:         }
405: 
406:         // pack into event required by filters and writers
407:         $event = $this->_packEvent($message, $priority);
408: 
409:         // Check to see if any extra information was passed
410:         if (!empty($extras)) {
411:             $info = array();
412:             if (is_array($extras)) {
413:                 foreach ($extras as $key => $value) {
414:                     if (is_string($key)) {
415:                         $event[$key] = $value;
416:                     } else {
417:                         $info[] = $value;
418:                     }
419:                 }
420:             } else {
421:                 $info = $extras;
422:             }
423:             if (!empty($info)) {
424:                 $event['info'] = $info;
425:             }
426:         }
427: 
428:         // abort if rejected by the global filters
429:         foreach ($this->_filters as $filter) {
430:             if (! $filter->accept($event)) {
431:                 return;
432:             }
433:         }
434: 
435:         // send to each writer
436:         foreach ($this->_writers as $writer) {
437:             $writer->write($event);
438:         }
439:     }

532:     /**
533:      * Register Logging system as an error handler to log php errors
534:      * Note: it still calls the original error handler if set_error_handler is able to return it.
535:      *
536:      * Errors will be mapped as:
537:      *   E_NOTICE, E_USER_NOTICE => NOTICE
538:      *   E_WARNING, E_CORE_WARNING, E_USER_WARNING => WARN
539:      *   E_ERROR, E_USER_ERROR, E_CORE_ERROR, E_RECOVERABLE_ERROR => ERR
540:      *   E_DEPRECATED, E_STRICT, E_USER_DEPRECATED => DEBUG
541:      *   (unknown/other) => INFO
542:      *
543:      * @link http://www.php.net/manual/en/function.set-error-handler.php Custom error handler
544:      *
545:      * @return Zend_Log
546:      */
547:     public function registerErrorHandler()
548:     {
549:         // Only register once.  Avoids loop issues if it gets registered twice.
550:         if ($this->_registeredErrorHandler) {
551:             return $this;
552:         }
553: 
554:         $this->_origErrorHandler = set_error_handler(array($this, 'errorHandler'));
555: 
556:         // Construct a default map of phpErrors to Zend_Log priorities.
557:         // Some of the errors are uncatchable, but are included for completeness
558:         $this->_errorHandlerMap = array(
559:             E_NOTICE            => Zend_Log::NOTICE,
560:             E_USER_NOTICE       => Zend_Log::NOTICE,
561:             E_WARNING           => Zend_Log::WARN,
562:             E_CORE_WARNING      => Zend_Log::WARN,
563:             E_USER_WARNING      => Zend_Log::WARN,
564:             E_ERROR             => Zend_Log::ERR,
565:             E_USER_ERROR        => Zend_Log::ERR,
566:             E_CORE_ERROR        => Zend_Log::ERR,
567:             E_RECOVERABLE_ERROR => Zend_Log::ERR,
568:             E_STRICT            => Zend_Log::DEBUG,
569:         );
