257:     {
258:         $this->scope++;
259:         $this->state = self::STATE_SENDING;
260:         $requestsInScope = empty($this->requests[$this->scope]) ? array() : $this->requests[$this->scope];
261: 
262:         // Only prepare and send requests that are in the current recursion scope
263:         // Only enter the main perform() loop if there are requests in scope
264:         if ($requestsInScope) {
265: 
266:             // Any exceptions thrown from this event should break the entire flow of sending requests
267:             $this->dispatch(self::BEFORE_SEND, array(
268:                 'requests' => $this->requests[$this->scope]
269:             ));
270: 
271:             foreach ($this->requests[$this->scope] as $request) {
272:                 if ($request->getState() != RequestInterface::STATE_TRANSFER) {
273:                     $this->beforeSend($request);
274:                 }
275:             }
276: 
277:             try {
278:                 $this->perform();

280:                 $this->exceptions[] = array('request' => null, 'exception' => $e);

286:         // Aggregate exceptions into a MultiTransferException if needed
287:         $multiException = $this->buildMultiTransferException($requestsInScope);

289:         // Complete the transfer if this is not a nested scope
290:         if ($this->scope == -1) {

296:         // Throw any exceptions that were encountered
297:         if ($multiException) {

324:         $store = new \SplObjectStorage();

326:             $store->attach($request);
327:         }

331:             $multiException->add($e['exception']);
332:             if (isset($e['request'])) {

334:                 // Remove from the total list so that it becomes a list of successful requests
335:                 unset($store[$e['request']]);

360:                 // Requests might decide they don't need to be sent just before transfer (e.g. CachePlugin)
361:                 $this->remove($request);

363:                 // Queued responses do not need to be sent using curl
364:                 $this->remove($request);
365:                 $request->setState(RequestInterface::STATE_COMPLETE);

367:                 // Add the request's curl handle to the multi handle
368:                 $this->checkCurlResult(curl_multi_add_handle($this->multiHandle, $this->createCurlHandle($request)->getHandle()));

391:     /**
392:      * Get the data from the multi handle
393:      */
394:     protected function perform()
395:     {
396:         // @codeCoverageIgnoreStart
397:         // Weird things can happen when making HTTP requests in __destruct methods
398:         if (!$this->multiHandle) {
399:             return;
400:         }
401:         // @codeCoverageIgnoreEnd
402: 
403:         // If there are no requests to send, then exit from the function
404:         if ($this->scope <= 0) {
405:             if ($this->count() == 0) {
406:                 return;
407:             }
408:         } elseif (empty($this->requests[$this->scope])) {
409:             return;
410:         }
411: 
412:         // Create the polling event external to the loop
413:         $event = array('curl_multi' => $this);
414:         $active = $this->executeHandles();
415: 
416:         while (1) {
417: 
418:             $this->processMessages();
419: 
420:             // Exit the function if there are no more requests to send
421:             if (!($scopedPolling = $this->scope <= 0 ? $this->all() : $this->requests[$this->scope])) {
422:                 break;
423:             }
424: 
425:             // Notify each request as polling
426:             $blocking = $total = 0;
427:             foreach ($scopedPolling as $request) {
428:                 $event['request'] = $request;
429:                 $request->dispatch(self::POLLING_REQUEST, $event);
430:                 ++$total;
431:                 // The blocking variable just has to be non-falsey to block the loop
432:                 if ($request->getParams()->hasKey(self::BLOCKING)) {
433:                     ++$blocking;
434:                 }
435:             }
436: 
437:             if ($blocking == $total) {
438:                 // Sleep to prevent eating CPU because no requests are actually pending a select call
439:                 usleep(500);
440:             } else {
441:                 // Select the curl handles until there is any activity on any of the open file descriptors
442:                 // See https://github.com/php/php-src/blob/master/ext/curl/multi.c#L170
443:                 $active = $this->executeHandles(true, 0.02, $active);
444:             }
445:         }
446:     }

449:      * Process any received curl multi messages

452:     {
453:         // Get messages from curl handles
454:         while ($done = curl_multi_info_read($this->multiHandle)) {
455:             try {
456:                 $request = $this->resourceHash[(int) $done['handle']];
457:                 $handle = $this->handles[$request];

465:     /**
466:      * Execute and select curl handles until there is activity
467:      *
468:      * @param bool $select  Set to TRUE to select the file descriptors
469:      * @param int  $timeout Select timeout in seconds
470:      * @param int  $active  Previous active value
471:      *
472:      * @return int Returns the number of active handles
473:      */
474:     private function executeHandles($select = false, $timeout = 1, $active = 0)
475:     {
476:         do {
477:             // @codeCoverageIgnoreStart
478:             if ($select && $active && curl_multi_select($this->multiHandle, $timeout) == -1) {
479:                 // Perform a usleep if a previously executed select returned -1
480:                 // @see https://bugs.php.net/bug.php?id=61141
481:                 usleep(125);
482:             }
483:             // @codeCoverageIgnoreEnd
484:             do {
485:                 $mrc = curl_multi_exec($this->multiHandle, $active);
486:             } while ($mrc == CURLM_CALL_MULTI_PERFORM);
487:             // Check the return value to ensure an error did not occur
488:             $this->checkCurlResult($mrc);
489:         // Poll once if not selecting, or poll until there are no handles with activity
490:         } while ($select && $active);
491: 
492:         return $active;
493:     }

512:     /**
513:      * Check for errors and fix headers of a request based on a curl response
514:      *
515:      * @param RequestInterface $request Request to process
516:      * @param CurlHandle       $handle  Curl handle object
517:      * @param array            $curl    Array returned from curl_multi_info_read
518:      *
519:      * @throws CurlException on Curl error
520:      */
521:     protected function processResponse(RequestInterface $request, CurlHandle $handle, array $curl)
522:     {
523:         // Set the transfer stats on the response
524:         $handle->updateRequestFromTransfer($request);
525:         // Check if a cURL exception occurred, and if so, notify things
526:         $curlException = $this->isCurlException($request, $handle, $curl);
527: 
528:         // Always remove completed curl handles.  They can be added back again
529:         // via events if needed (e.g. ExponentialBackoffPlugin)
530:         $this->removeHandle($request);
531: 
532:         if (!$curlException) {
533:             $request->setState(RequestInterface::STATE_COMPLETE, array('handle' => $handle));
534:             // Only remove the request if it wasn't resent as a result of the state change
535:             if ($request->getState() != RequestInterface::STATE_TRANSFER) {
536:                 $this->remove($request);
537:             }
538:         } else {
539:             // Set the state of the request to an error
540:             $request->setState(RequestInterface::STATE_ERROR);
541:             // Notify things that listen to the request of the failure
542:             $request->dispatch('request.exception', array(
543:                 'request'   => $this,
544:                 'exception' => $curlException
545:             ));
546: 
547:             // Allow things to ignore the error if possible
548:             $state = $request->getState();
549:             if ($state != RequestInterface::STATE_TRANSFER) {
550:                 $this->remove($request);
551:             }
552:             // The error was not handled, so fail
553:             if ($state == RequestInterface::STATE_ERROR) {
554:                 /** @var $curlException \Exception */
555:                 throw $curlException;
556:             }
557:         }
558:     }

560:     /**
561:      * Remove a curl handle from the curl multi object
562:      *
563:      * Nasty things (bus errors, segmentation faults) can sometimes occur when removing curl handles when in a callback
564:      * or a recursive scope.  Here we are queueing all curl handles that need to be removed and closed so that this
565:      * happens only in the outermost scope when everything has completed sending.
566:      *
567:      * @param RequestInterface $request Request that owns the handle
568:      */

570:     {

572:             $handle = $this->handles[$request];
573:             unset($this->resourceHash[(int) $handle->getHandle()]);
574:             unset($this->handles[$request]);
575:             $this->removeHandles[] = $handle;
576:         }
577:     }
