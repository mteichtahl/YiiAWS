112:         $longDate = $this->getDateTime(DateFormat::ISO8601);
113:         $shortDate = $this->getDateTime(DateFormat::SHORT);

115:         // Remove any previously set Authorization headers so that
116:         // exponential backoff works correctly
117:         $request->removeHeader('Authorization');

119:         // Requires a x-amz-date header or Date
120:         if ($request->hasHeader('x-amz-date') || !$request->hasHeader('Date')) {

122:         } else {
123:             $request->setHeader('Date', $this->getDateTime(DateFormat::RFC1123));

126:         // Add the security token if one is present

131:         // Parse the region and service name from the request URL
132:         $url = Url::factory($request->getUrl());

134:         // Parse the service and region or use one that is explicitly set
135:         $region = $this->regionName ?: HostNameUtils::parseRegionName($url);

159:     /**
160:      * Create the canonical representation of a request
161:      *
162:      * @param RequestInterface $request Request to canonicalize
163:      *
164:      * @return string
165:      */
166:     private function createCanonicalRequest(RequestInterface $request)
167:     {
168:         // Normalize the path as required by SigV4
169:         $path = $request->getUrl(true)->normalizePath()->getPath();
170: 
171:         $canon = $request->getMethod() . "\n{$path}\n"
172:             . $this->getCanonicalizedQueryString($request) . "\n";
173: 
174:         // Create the canonical headers
175:         $headers = array();
176:         foreach ($request->getHeaders() as $key => $values) {
177:             $key = strtolower($key);
178:             if (!isset($headers[$key])) {
179:                 $headers[$key] = array();
180:             }
181:             foreach ($values as $value) {
182:                 $headers[$key][] = preg_replace('/\s+/', ' ', trim($value));
183:             }
184:         }
185: 
186:         // The headers must be sorted
187:         ksort($headers);
188: 
189:         // Continue to build the canonical request by adding headers
190:         foreach ($headers as $key => $values) {
191:             // Combine multi-value headers into a sorted comma separated list
192:             if (count($values) > 1) {
193:                 sort($values);
194:             }
195:             $value = implode(',', $values);
196:             $canon .= $key . ':' . $value . "\n";
197:         }
198: 
199:         // Create the signed headers
200:         $signedHeaders = implode(';', array_keys($headers));
201:         $canon .= "\n{$signedHeaders}\n";
202:         $request->getParams()->set('aws.signed_headers', $signedHeaders);
203: 
204:         // Create the payload if this request has an entity body
205:         if ($request->hasHeader('x-amz-content-sha256')) {
206:             // Handle streaming operations (e.g. Glacier.UploadArchive)
207:             $canon .= $request->getHeader('x-amz-content-sha256');
208:         } elseif ($request instanceof EntityEnclosingRequestInterface) {
209:             $canon .= hash(
210:                 'sha256',
211:                 $request->getMethod() == 'POST' && count($request->getPostFields())
212:                     ? (string) $request->getPostFields() : (string) $request->getBody()
213:             );
214:         } else {
215:             $canon .= self::DEFAULT_PAYLOAD;
216:         }
217: 
218:         // Add debug information
219:         $request->getParams()->set('aws.canonical_request', $canon);
220: 
221:         return $canon;
222:     }
